2.1 base

2.1.1 funcões

  • função com um argumento

    ::: {.cell hash=‘mod0_02_intro_r_cache/html/unnamed-chunk-1_f4bb63b72a3e25b20071d819429f12c4’}

    # declarando
atualizar <- function(histórico) {
    atual <- ((histórico + 17)/2)
    return(atual)
}
# usando um escalar
atualizar(1)
# usando com um vetor
atualizar(c(1:15))

    :::

  • função com dois argumentos

    ::: {.cell hash=‘mod0_02_intro_r_cache/html/unnamed-chunk-2_4d8e2f7824558e4d45bbb07239555611’}

    # declarando
ajustar <- function(valor1, valor2) {
    ajuste <- ((valor1 + 180)/(valor2 - 60))
    return(ajuste)
}
# usando com escalares
ajustar(100, 80)
# usando com vetores
ajustar(c(0,1,2), c(61, 62, 63)) 
# usando com vetor  e escalar
ajustar(c(0,1,2), 61)

    :::

2.1.2 estruturas condicionais

  • if

    ::: {.cell hash=‘mod0_02_intro_r_cache/html/unnamed-chunk-3_e88cc3d77a2964054687a2ed6c9a8b1f’}

    valor <- 100000
if (valor >= 1000000) {
    "número grande"
}

    :::

  • if e else

    ::: {.cell hash=‘mod0_02_intro_r_cache/html/unnamed-chunk-4_80b7a9cf3483df9bd65c5ca3a4556ded’}

    valor <- 100000
if (valor >= 1000000) {
    ret <- "número grande"
} else {
    ret <- "número pequeno"
}

    :::

  • ternário

    ::: {.cell hash=‘mod0_02_intro_r_cache/html/unnamed-chunk-5_7f180ee9540ce826385bea3478363a18’}

    valor <- 100000
ret <- if (valor >= 1000000) "número grande" else "número pequeno"

    :::

  • múltiplas condições

    ::: {.cell hash=‘mod0_02_intro_r_cache/html/unnamed-chunk-6_bc79d776c8b34bba15c51835b391ce1e’}

    valor <- 650000

if (valor >= 1000000) {
    ret <- "número grande"
} else if (valor >= 500000 & valor <1000000) {
    ret <- "número intermediário"
} else {
    ret <- "número pequeno"
}

    :::
